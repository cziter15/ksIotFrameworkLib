27namespace ksf::evt
28{
29 template <typename... Params>
30 class ksEvent : public evt::ksEventInterface
31 {
32 protected:
33 std::vector<std::pair<std::size_t, std::function<void(Params...)>>> callbacks; // List of bond callbacks.
34 std::size_t lastCallbackUID{0}; // Last unique callback ID for this event (used as counter).
35
36 public:
37 /*
38 @brief Returns whether any callback is bound to this event.
39 @return True if any callback is bound to this event. False otherwise.
40 */
41 bool isBound() const
42 {
43 return !callbacks.empty();
44 }
45
46 /*
47 @brief Registers event (binds to callback list).
48 @param outHandle Reference to outHandle shared ptr (destruction of ksEventHandle object will unbind the event)
49 @param function R-value reference to callback function
50 */
51 void registerEvent(std::unique_ptr<ksf::evt::ksEventHandle>& outHandle, std::function<void(Params...)>&& function)
52 {
53 ++lastCallbackUID;
54 outHandle = std::make_unique<ksf::evt::ksEventHandle>(weak_from_this(), lastCallbackUID);
55 callbacks.emplace_back(lastCallbackUID, std::move(function));
56 }
57
58 /*
59 @brief Unbinds event callback by specified unique ID.
60
61 This ID is automatically assigned by registerEvent function and
62 ksEventHandle uses it to unbind automatically upon destruction.
63
64 @param callbackUID Unique id of the callback to unbind.
65 */
66 void unbind(std::size_t callbackUID) override
67 {
68 auto predicate = [callbackUID](const auto& cb) {
69 auto& [uid, function] = cb;
70 return uid == callbackUID;
71 };
72
73 callbacks.erase(std::remove_if(callbacks.begin(), callbacks.end(), predicate), callbacks.end());
74 }
75
76 /*
77 @brief Broadcasts event to all bound callbacks.
78 @param args Event parameters.
79 */
80 void broadcast(Params... params)
81 {
82 for (const auto& [uid, function] : callbacks)
83 function(params...);
84 }
85 };
86}
