163 inline std::string to_hex(int value)
164 {
165 if (value == 0)
166 return {"0"};
167
168 char buffer[8];
169 unsigned int uvalue{static_cast<unsigned int>(value)}, pos{8};
170
171 while (uvalue != 0)
172 {
173 unsigned int nibble{uvalue & 0xf};
174 buffer[--pos] = (nibble < 10) ? ('0' + nibble) : ('a' + (nibble - 10));
175 uvalue >>= 4;
176 }
177
178 return std::string(buffer + pos, 8 - pos);
179 }
